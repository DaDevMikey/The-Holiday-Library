Quick Start

// Add to your HTML <script src="https://dadevmikey.github.io/The-Holiday-Library/HalloweenOverlay.js"></script> // Initialize and start Halloween effects const halloween = new HalloweenOverlay(); halloween.startAnimation(); // Or customize it const halloween = new HalloweenOverlay({ pumpkinCount: 8, batCount: 12, ghostCount: 10, leafCount: 20, spiderCount: 5, enableLightning: true, enableFog: true, enableSpiders: true, pumpkinColors: ['#FF6B1A', '#FF8C00', '#FFA500'], batColors: ['#000000', '#1a0033', '#2d0066'] }); halloween.startAnimation(); // Stop Halloween effects halloween.stopAnimation();

API Options

Basic Usage

const hw = new HalloweenOverlay(); hw.startAnimation(); hw.stopAnimation(); hw.toggle();

Element Counts

new HalloweenOverlay({ pumpkinCount: 8, batCount: 12, ghostCount: 10, leafCount: 20, spiderCount: 5 });

Color Customization

new HalloweenOverlay({ pumpkinColors: ['#FF6B1A', '#FF8C00'], batColors: ['#000000', '#1a0033'], ghostColors: ['rgba(255,255,255,0.8)'], leafColors: ['#8B4513', '#FF8C00'] });

Effect Toggles

new HalloweenOverlay({ enableLightning: true, enableFog: true, enableSpiders: true });

Animation Speeds

new HalloweenOverlay({ pumpkinSpeed: { min: 1, max: 3 }, batSpeed: { min: 2, max: 5 }, ghostSpeed: { min: 0.5, max: 1.5 }, leafSpeed: { min: 1, max: 3 } });

Lightning Settings

new HalloweenOverlay({ enableLightning: true, lightningInterval: { min: 8000, max: 20000 }, lightningDuration: 150 });

Time-Based Scheduling

const hw = new HalloweenOverlay({ startTime: '18:00', endTime: '23:59' }); hw.startAnimation();

Date-Based Scheduling

new HalloweenOverlay({ dateStart: '10-25', // Oct 25 dateEnd: '11-01' // Nov 1 });

Auto-Stop Timeout

new HalloweenOverlay({ timeout: 60000 // 60 seconds });
